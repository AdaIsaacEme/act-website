### With Error Handling

```tsx
<img
  src="/images/projects/project.jpg"
  alt="Project"
  onError={(e) => {
    e.currentTarget.style.display = "none";
  }}
/>
```

### Using ImageWithFallback Component

```tsx
import ImageWithFallback from "../components/ImageWithFallback";

<ImageWithFallback
  src="/images/projects/project.jpg"
  alt="Project"
  fallbackSrc="/images/projects/default.jpg"
/>;
```
